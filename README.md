# github-guide
-- Create new repository
git init
git commit -m "first commit"
git branch -m main
git remote add origin/github https://github.com/nhoaghai/github-guide.git
git push -u origin/github main

-- Push an existing repository
git remote add origin/github https://github.com/nhoaghai/github-guide.git
git branch -m main
git push -u origin/github main
